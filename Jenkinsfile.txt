pipeline {
    agent any
    stages {
        stage('Chmod') {
            steps {
                sh 'chmod +x gradlew'
            }
        }
        stage('Gradle Build') {
          steps {
            sh './gradlew build'
          }
        }
        stage('Test') {
          steps {
            sh './gradlew clean test'
          }
        }
    }
}
