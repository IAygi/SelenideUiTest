pipeline {
    agent any
    stages {
        stage('Chmod') {
            steps {
                sh 'chmod +x gradlew'
            }
        }
        stage('Gradle Build') {
          steps {
            sh './gradlew build --no-daemon'
          }
        }
        stage('Test') {
          steps {
            sh './gradlew clean test --no-daemon'
          }
        }
        stage('Allure Serve') {
          steps {
            allure([
            includeProperties: false,
            jdk: '',
            properties: [],
            reportBuildPolicy: 'ALWAYS',
            results: [[path: 'allure-results']]
            ])
          }
        }
    }
}
