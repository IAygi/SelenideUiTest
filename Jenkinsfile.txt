pipeline {
    agent any
    stages {
        stage('Chmod') {
            steps {
                sh 'chmod +x gradlew'
            }
        }
        stage('Gradle Build') {
          steps {
            sh './gradlew build'
          }
        }
        stage('Test') {
          steps {
            sh './gradlew clean test'
          }
        }
    }
    post {
            success { allure([
            includeProperties: false,
            jdk: '',
            properties: [],
            reportBuildPolicy: 'ALWAYS',
            results: [[path: 'target/allure-results']]
            ])
        }
     }
}
